BEGIN
    DECLARE array judulBuku
    DECLARE array stokBuku
    DECLARE integer jumlahBuku = 0
    DECLARE integer menu

    DO
        DISPLAY "Menu Perpustakaan"
        DISPLAY "1. Input Buku"
        DISPLAY "2. Tampilkan Buku"
        DISPLAY "3. Pinjam Buku"
        DISPLAY "4. Keluar"

        INPUT menu

        IF menu == 1 THEN
            INPUT judulBuku[jumlahBuku]
            INPUT stokBuku[jumlahBuku]
            jumlahBuku = jumlahBuku + 1

        ELSE IF menu == 2 THEN
            IF jumlahBuku == 0 THEN
                DISPLAY "Belum ada data buku"
            ELSE
                FOR i = 0 TO jumlahBuku - 1 DO
                    DISPLAY judulBuku[i] dan stokBuku[i]
                END FOR
            END IF

        ELSE IF menu == 3 THEN
            INPUT nomorBuku

            IF nomorBuku >= 1 AND nomorBuku <= jumlahBuku THEN
                IF stokBuku[nomorBuku - 1] > 0 THEN
                    stokBuku[nomorBuku - 1] = stokBuku[nomorBuku - 1] - 1
                    DISPLAY "Buku berhasil dipinjam"
                ELSE
                    DISPLAY "Stok buku habis"
                END IF
            ELSE
                DISPLAY "Nomor buku tidak valid"
            END IF

        END IF

    WHILE menu != 4

    DISPLAY "Program selesai"
END
